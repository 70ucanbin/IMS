create table "public".com_employee (
  employee_id character varying(20)
  , user_id character varying(20)
  , user_name character varying(20)
  , password character varying(50)
  , email character varying(50)
  , PRIMARY KEY (employee_id, user_id)
);

create table "public".com_item (
  item_category character varying(2)
  , item_key character varying(20)
  , item_value character varying(100)
  , display_order integer
  , PRIMARY KEY (item_category, item_key)
);

create table "public".tra_attached_file (
  attached_file_id character varying(20)
  , file_name character varying(20)
  , file_size integer
  , file_data bytea
  , PRIMARY KEY (attached_file_id)
);

create table "public".tra_client_work (
  employee_id character varying(20)
  , work_year smallint
  , work_month smallint
  , work_day smallint
  , order_cd character varying(20)
  , task_cd character varying(20)
  , sub_order_cd character varying(20)
  , hours_of_work smallint
  , minutes_of_work smallint
  , note character varying(200)
  , PRIMARY KEY (employee_id, work_year, work_month)
);

create table "public".tra_monthly_report (
  employee_id character varying(20)
  , work_year smallint
  , work_month smallint
  , work_day smallint
  , work_details character varying(100)
  , start_work_hours smallint
  , start_work_minutes smallint
  , end_work_hours smallint
  , end_work_minutes smallint
  , normal_working_hours numeric(3, 1)
  , overtime_hours numeric(3, 1)
  , holiday_work_hours numeric(3, 1)
  , note character varying(200)
  , PRIMARY KEY (employee_id, work_year, work_month, work_day)
);

create table "public".tra_travel_expenses (
  travel_expenses_id integer
  , employee_id character varying(20)
  , work_year smallint
  , work_month smallint
  , expense_date character varying(30)
  , expense_item character varying(50)
  , route character varying(50)
  , transit character varying(50)
  , payment integer
  , attached_file_id character varying(20)
  , note character varying(200)
  , PRIMARY KEY (travel_expenses_id)
);

CREATE SEQUENCE tra_id_seq START 1;
